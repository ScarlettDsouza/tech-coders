//Program to find sum and average of 3 integers

#include <stdio.h>

int main()
{
    int num1, num2, num3, sum;
    float avg;
    scanf("%d %d %d", &num1, &num2, &num3);
    sum = num1 + num2 + num3;
    avg = sum / 3.0; // avg = (float)sum / 3;
    
    printf("Sum : %d\n", sum);
    printf("Average : %g", avg);

    return 0;
}
