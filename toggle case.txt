//Program to convert a character into toggle case
#include <stdio.h>

int main()
{
    char ch;
    ch = getc(stdin);
    if ( ch >='a' && ch <='z')
        printf("Upper case '%c' is '%c'", ch, ch-32);
    else if ( ch >='A' && ch <='Z')
        printf("Lower case '%c' is '%c'", ch, ch+32);
    else
        printf("Not an alphabet");
    return 0;
}
